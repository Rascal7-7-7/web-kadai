サンプルコード

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Hello World
int main() {
    printf("Hello, World!\n");
    
    // 変数の宣言と使用
    int number = 42;
    char message[] = "C言語";
    float pi = 3.14159;
    
    printf("%s - %d - %.2f\n", message, number, pi);
    
    // 配列の使用
    int numbers[] = {1, 2, 3, 4, 5};
    int size = sizeof(numbers) / sizeof(numbers[0]);
    
    for (int i = 0; i < size; i++) {
        printf("Number[%d]: %d\n", i, numbers[i]);
    }
    
    // 関数の呼び出し
    int result = add(10, 20);
    printf("10 + 20 = %d\n", result);
    
    // ポインタの使用
    int value = 100;
    int *ptr = &value;
    printf("Value: %d, Address: %p\n", *ptr, ptr);
    
    // 動的メモリ割り当て
    int *dynamic_array = malloc(5 * sizeof(int));
    if (dynamic_array != NULL) {
        for (int i = 0; i < 5; i++) {
            dynamic_array[i] = i * 10;
            printf("Dynamic[%d]: %d\n", i, dynamic_array[i]);
        }
        free(dynamic_array);
    }
    
    return 0;
}

// 関数の定義
int add(int a, int b) {
    return a + b;
}
