サンプルコード

       IDENTIFICATION DIVISION.
       PROGRAM-ID. HELLO-WORLD.
       
       ENVIRONMENT DIVISION.
       CONFIGURATION SECTION.
       
       DATA DIVISION.
       WORKING-STORAGE SECTION.
       01 GREETING        PIC X(20) VALUE "HELLO, WORLD!".
       01 CURRENT-DATE-DATA.
          05 CURRENT-DATE.
             10 CURRENT-YEAR   PIC 9(4).
             10 CURRENT-MONTH  PIC 9(2).
             10 CURRENT-DAY    PIC 9(2).
          05 CURRENT-TIME.
             10 CURRENT-HOUR   PIC 9(2).
             10 CURRENT-MINUTE PIC 9(2).
             10 CURRENT-SECOND PIC 9(2).
             10 CURRENT-MS     PIC 9(2).
       
       01 EMPLOYEE-RECORD.
          05 EMPLOYEE-ID       PIC 9(5).
          05 EMPLOYEE-NAME     PIC X(30).
          05 EMPLOYEE-DEPT     PIC X(20).
          05 EMPLOYEE-SALARY   PIC 9(7)V99.
       
       01 COUNTER             PIC 9(3) VALUE 0.
       01 TOTAL-SALARY        PIC 9(9)V99 VALUE 0.
       01 AVERAGE-SALARY      PIC 9(7)V99.
       
       PROCEDURE DIVISION.
       MAIN-PROCEDURE.
           DISPLAY GREETING.
           
           MOVE FUNCTION CURRENT-DATE TO CURRENT-DATE-DATA.
           DISPLAY "現在日時: " CURRENT-YEAR "/" CURRENT-MONTH "/"
                  CURRENT-DAY " " CURRENT-HOUR ":" CURRENT-MINUTE.
           
           PERFORM INITIALIZE-EMPLOYEE.
           PERFORM DISPLAY-EMPLOYEE.
           
           PERFORM PROCESS-EMPLOYEES.
           
           DISPLAY "処理完了".
           STOP RUN.
       
       INITIALIZE-EMPLOYEE.
           MOVE 12345 TO EMPLOYEE-ID.
           MOVE "山田太郎" TO EMPLOYEE-NAME.
           MOVE "開発部" TO EMPLOYEE-DEPT.
           MOVE 450000.00 TO EMPLOYEE-SALARY.
       
       DISPLAY-EMPLOYEE.
           DISPLAY "社員情報:".
           DISPLAY "  ID: " EMPLOYEE-ID.
           DISPLAY "  名前: " EMPLOYEE-NAME.
           DISPLAY "  部署: " EMPLOYEE-DEPT.
           DISPLAY "  給与: " EMPLOYEE-SALARY.
       
       PROCESS-EMPLOYEES.
           PERFORM VARYING COUNTER FROM 1 BY 1 UNTIL COUNTER > 5
              DISPLAY "処理中: " COUNTER
              ADD EMPLOYEE-SALARY TO TOTAL-SALARY
           END-PERFORM.
           
           COMPUTE AVERAGE-SALARY = TOTAL-SALARY / 5.
           DISPLAY "合計給与: " TOTAL-SALARY.
           DISPLAY "平均給与: " AVERAGE-SALARY.
           
           EVALUATE AVERAGE-SALARY
              WHEN 0 THRU 300000.00
                 DISPLAY "給与レベル: 低"
              WHEN 300000.01 THRU 600000.00
                 DISPLAY "給与レベル: 中"
              WHEN OTHER
                 DISPLAY "給与レベル: 高"
           END-EVALUATE.
