サンプルコード

#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

// クラスの定義
class Person {
private:
    string name;
    int age;

public:
    // コンストラクタ
    Person(const string& n, int a) : name(n), age(a) {}
    
    // ゲッター
    string getName() const { return name; }
    int getAge() const { return age; }
    
    // メソッド
    void introduce() const {
        cout << "こんにちは、" << name << "です。" << age << "歳です。" << endl;
    }
};

// テンプレート関数
template<typename T>
T maximum(T a, T b) {
    return (a > b) ? a : b;
}

int main() {
    cout << "Hello, C++!" << endl;
    
    // オブジェクトの作成
    Person person("田中太郎", 25);
    person.introduce();
    
    // STL vectorの使用
    vector<int> numbers = {5, 2, 8, 1, 9, 3};
    
    // ソート
    sort(numbers.begin(), numbers.end());
    
    cout << "ソート後: ";
    for (int num : numbers) {
        cout << num << " ";
    }
    cout << endl;
    
    // テンプレート関数の使用
    cout << "最大値: " << maximum(10, 20) << endl;
    cout << "最大値: " << maximum(3.14, 2.71) << endl;
    
    // ラムダ式
    auto square = [](int x) { return x * x; };
    cout << "5の二乗: " << square(5) << endl;
    
    // STL アルゴリズム
    vector<int> squares;
    transform(numbers.begin(), numbers.end(), back_inserter(squares), square);
    
    cout << "二乗値: ";
    for (int sq : squares) {
        cout << sq << " ";
    }
    cout << endl;
    
    return 0;
}
