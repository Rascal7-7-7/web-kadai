// MongoDB Shell コマンド

// データベース作成・選択
use myDatabase

// コレクション作成
db.createCollection("users")
db.createCollection("products")

// ドキュメント挿入
db.users.insertOne({
  name: "田中太郎",
  email: "tanaka@example.com",
  age: 30,
  address: {
    city: "東京",
    zipcode: "100-0001"
  },
  interests: ["プログラミング", "読書", "旅行"],
  created_at: new Date()
})

// 複数ドキュメント挿入
db.products.insertMany([
  {
    name: "ノートパソコン",
    price: 120000,
    category: "電子機器",
    stock: 10,
    specs: {
      cpu: "Core i7",
      ram: "16GB",
      storage: "512GB SSD"
    }
  },
  {
    name: "スマートフォン",
    price: 80000,
    category: "電子機器",
    stock: 15,
    specs: {
      screen: "6.1インチ",
      camera: "12MP",
      storage: "128GB"
    }
  },
  {
    name: "ヘッドフォン",
    price: 15000,
    category: "アクセサリー",
    stock: 20
  }
])

// ドキュメント検索
// 全件取得
db.users.find()

// 条件指定
db.products.find({ category: "電子機器" })

// 複合条件
db.products.find({
  category: "電子機器",
  price: { $gt: 100000 }
})

// プロジェクション（フィールド選択）
db.products.find(
  { category: "電子機器" },
  { name: 1, price: 1, _id: 0 }
)

// ソート
db.products.find().sort({ price: -1 })

// 制限
db.products.find().limit(2)

// スキップ
db.products.find().skip(1).limit(2)

// ドキュメント更新
// 単一ドキュメント更新
db.products.updateOne(
  { name: "ノートパソコン" },
  { $set: { price: 110000, stock: 8 } }
)

// 複数ドキュメント更新
db.products.updateMany(
  { category: "電子機器" },
  { $inc: { stock: -1 } }
)

// ドキュメント削除
db.products.deleteOne({ name: "ヘッドフォン" })
db.products.deleteMany({ stock: { $lte: 0 } })

// インデックス作成
db.products.createIndex({ name: 1 })
db.products.createIndex({ category: 1, price: -1 })

// 複合クエリ
db.products.find({
  $and: [
    { category: "電子機器" },
    { price: { $gte: 50000 } },
    { stock: { $gt: 0 } }
  ]
})

// OR条件
db.products.find({
  $or: [
    { category: "電子機器" },
    { price: { $lt: 20000 } }
  ]
})

// 正規表現検索
db.products.find({
  name: { $regex: "フォン", $options: "i" }
})

// 配列検索
db.users.find({
  interests: "プログラミング"
})

// ネストされたフィールド検索
db.users.find({
  "address.city": "東京"
})

// アグリゲーション
db.products.aggregate([
  { $match: { category: "電子機器" } },
  { $group: {
      _id: "$category",
      count: { $sum: 1 },
      avgPrice: { $avg: "$price" },
      totalStock: { $sum: "$stock" }
    }
  }
])

// 複雑なアグリゲーションパイプライン
db.products.aggregate([
  { $match: { stock: { $gt: 0 } } },
  { $group: {
      _id: "$category",
      count: { $sum: 1 },
      avgPrice: { $avg: "$price" },
      items: { $push: { name: "$name", price: "$price" } }
    }
  },
  { $sort: { avgPrice: -1 } }
])

// Node.jsでのMongoDBの使用例
// MongoDB Node.js Driver
const { MongoClient } = require('mongodb');

// 接続URL
const url = 'mongodb://localhost:27017';
const client = new MongoClient(url);

// データベース名
const dbName = 'myDatabase';

async function main() {
  try {
    // 接続
    await client.connect();
    console.log('MongoDBに接続しました');
    
    const db = client.db(dbName);
    const usersCollection = db.collection('users');
    
    // ドキュメント挿入
    const insertResult = await usersCollection.insertOne({
      name: '佐藤花子',
      email: 'sato@example.com',
      age: 25
    });
    console.log('挿入されたID:', insertResult.insertedId);
    
    // ドキュメント検索
    const findResult = await usersCollection.find({}).toArray();
    console.log('検索結果:', findResult);
    
    // ドキュメント更新
    const updateResult = await usersCollection.updateOne(
      { name: '佐藤花子' },
      { $set: { age: 26 } }
    );
    console.log('更新されたドキュメント数:', updateResult.modifiedCount);
    
    // ドキュメント削除
    const deleteResult = await usersCollection.deleteOne({ name: '佐藤花子' });
    console.log('削除されたドキュメント数:', deleteResult.deletedCount);
    
  } finally {
    // 接続を閉じる
    await client.close();
  }
}

main().catch(console.error);

// Mongoose (ODM) の使用例
const mongoose = require('mongoose');

// 接続
mongoose.connect('mongodb://localhost:27017/myDatabase', {
  useNewUrlParser: true,
  useUnifiedTopology: true
});

// スキーマ定義
const userSchema = new mongoose.Schema({
  name: { type: String, required: true },
  email: { type: String, required: true, unique: true },
  age: { type: Number, min: 0 },
  address: {
    city: String,
    zipcode: String
  },
  interests: [String],
  created_at: { type: Date, default: Date.now }
});

// モデル作成
const User = mongoose.model('User', userSchema);

// ドキュメント作成と保存
async function createUser() {
  try {
    const user = new User({
      name: '山田太郎',
      email: 'yamada@example.com',
      age: 35,
      address: {
        city: '大阪',
        zipcode: '530-0001'
      },
      interests: ['スポーツ', '音楽']
    });
    
    const savedUser = await user.save();
    console.log('保存されたユーザー:', savedUser);
  } catch (err) {
    console.error('エラー:', err);
  }
}

// ドキュメント検索
async function findUsers() {
  try {
    // 全件取得
    const allUsers = await User.find();
    console.log('全ユーザー:', allUsers);
    
    // 条件指定
    const youngUsers = await User.find({ age: { $lt: 30 } })
      .select('name email age')
      .sort({ age: 1 });
    console.log('若いユーザー:', youngUsers);
  } catch (err) {
    console.error('エラー:', err);
  }
}

createUser();
findUsers();
