サンプルコード

# Hello World
print("Hello, World!")

# 変数と基本的な操作
name = "Python"
version = 3.9
is_awesome = True

print(f"{name} {version} is awesome: {is_awesome}")

# リストと繰り返し
numbers = [1, 2, 3, 4, 5]
for num in numbers:
    print(f"Number: {num}")

# 関数の定義
def greet(name):
    return f"Hello, {name}!"

print(greet("プログラマー"))

# 辞書の使用
person = {
    "name": "Alice",
    "age": 30,
    "city": "Tokyo"
}

for key, value in person.items():
    print(f"{key}: {value}")
