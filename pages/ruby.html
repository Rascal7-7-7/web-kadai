サンプルコード

# Hello World
puts "Hello, World!"

# 変数の使用
message = "Ruby言語"
number = 42
pi = 3.14159

puts "#{message} - #{number} - #{pi}"

# 配列の使用
numbers = [1, 2, 3, 4, 5]
numbers.each_with_index do |num, index|
  puts "Number[#{index}]: #{num}"
end

# ハッシュ（連想配列）
person = {
  name: "田中太郎",
  age: 25,
  city: "東京"
}

person.each do |key, value|
  puts "#{key}: #{value}"
end

# メソッドの定義
def add(a, b)
  a + b
end

result = add(10, 20)
puts "10 + 20 = #{result}"

# クラスの定義
class Person
  attr_accessor :name, :age
  
  def initialize(name, age)
    @name = name
    @age = age
  end
  
  def introduce
    "こんにちは、#{@name}です。#{@age}歳です。"
  end
  
  def adult?
    @age >= 18
  end
end

# オブジェクトの作成
person = Person.new("佐藤花子", 30)
puts person.introduce
puts "成人: #{person.adult?}"

# ブロックとイテレータ
(1..5).each { |i| puts "数字: #{i}" }

# 配列の操作
numbers = [1, 2, 3, 4, 5]
squares = numbers.map { |n| n * n }
puts "二乗: #{squares}"

evens = numbers.select { |n| n.even? }
puts "偶数: #{evens}"

# 文字列操作
text = "ruby programming"
puts text.upcase
puts text.capitalize
puts text.split.map(&:capitalize).join(' ')

# ファイル操作
File.open('sample.txt', 'w') do |file|
  file.puts "Rubyでファイルに書き込み"
  file.puts "ブロックを使用した安全な操作"
end

content = File.read('sample.txt')
puts "ファイル内容:\n#{content}"

# 例外処理
begin
  result = 10 / 0
rescue ZeroDivisionError => e
  puts "エラー: #{e.message}"
ensure
  puts "処理完了"
end

# モジュールとMixin
module Greetable
  def greet
    "Hello from #{self.class.name}!"
  end
end

class Robot
  include Greetable
  
  def initialize(name)
    @name = name
  end
end

robot = Robot.new("R2D2")
puts robot.greet
