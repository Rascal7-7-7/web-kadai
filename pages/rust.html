サンプルコード

// Hello World
fn main() {
    println!("Hello, World!");

    // 変数の宣言
    let message = "Rust言語";
    let mut number = 42;
    let pi = 3.14159;

    println!("{} - {} - {}", message, number, pi);

    // ベクタの使用
    let numbers = vec![1, 2, 3, 4, 5];
    for (index, num) in numbers.iter().enumerate() {
        println!("Number[{}]: {}", index, num);
    }

    // 関数の呼び出し
    let result = add(10, 20);
    println!("10 + 20 = {}", result);

    // 構造体の使用
    let person = Person::new("田中太郎".to_string(), 25);
    println!("{}", person.introduce());

    // 所有権の移動
    let s1 = String::from("Hello");
    let s2 = s1; // s1の所有権がs2に移動
    // println!("{}", s1); // これはエラーなのでコメントアウト
    println!("{}", s2);

    // 借用（参照）
    let s3 = String::from("World");
    let len = calculate_length(&s3);
    println!("'{}' の長さは {} です", s3, len);

    // Option型
    let some_number = Some(5);
    let no_number: Option = None;

    match some_number {
        Some(value) => println!("値: {}", value),
        None => println!("値なし"),
    }

    // Result型とエラーハンドリング
    match divide(10, 2) {
        Ok(result) => println!("結果: {}", result),
        Err(error) => println!("エラー: {}", error),
    }

    // イテレータとクロージャ
    let squares: Vec = numbers.iter().map(|x| x * x).collect();
    println!("二乗: {:?}", squares);

    let evens: Vec<&i32> = numbers.iter().filter(|&x| x % 2 == 0).collect();
    println!("偶数: {:?}", evens);
}

// 関数の定義
fn add(a: i32, b: i32) -> i32 {
    a + b
}

// 構造体の定義
struct Person {
    name: String,
    age: u32,
}

impl Person {
    fn new(name: String, age: u32) -> Person {
        Person { name, age }
    }
    fn introduce(&self) -> String {
        format!("こんにちは、{}です。{}歳です。", self.name, self.age)
    }
    fn is_adult(&self) -> bool {
        self.age >= 18
    }
}

// 借用を使った関数
fn calculate_length(s: &String) -> usize {
    s.len()
}

// Result型を返す関数
fn divide(a: i32, b: i32) -> Result {
    if b == 0 {
        Err("ゼロで割ることはできません".to_string())
    } else {
        Ok(a / b)
    }
}
