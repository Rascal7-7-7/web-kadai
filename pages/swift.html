サンプルコード

import Foundation

// Hello World
print("Hello, World!")

// 変数と定数
var message = "Swift言語"
let number = 42
let pi = 3.14159

print("\(message) - \(number) - \(pi)")

// 配列の使用
var numbers = [1, 2, 3, 4, 5]
for (index, num) in numbers.enumerated() {
    print("Number[\(index)]: \(num)")
}

// 辞書の使用
var person = [
    "name": "田中太郎",
    "age": "25",
    "city": "東京"
]

for (key, value) in person {
    print("\(key): \(value)")
}

// 関数の定義
func add(_ a: Int, _ b: Int) -> Int {
    return a + b
}

let result = add(10, 20)
print("10 + 20 = \(result)")

// クラスの定義
class Person {
    var name: String
    var age: Int
    
    init(name: String, age: Int) {
        self.name = name
        self.age = age
    }
    
    func introduce() -> String {
        return "こんにちは、\(name)です。\(age)歳です。"
    }
    
    func isAdult() -> Bool {
        return age >= 18
    }
}

// オブジェクトの作成
let person1 = Person(name: "佐藤花子", age: 30)
print(person1.introduce())
print("成人: \(person1.isAdult())")

// 構造体の定義
struct Point {
    var x: Double
    var y: Double
    
    func distance(to other: Point) -> Double {
        let dx = x - other.x
        let dy = y - other.y
        return sqrt(dx * dx + dy * dy)
    }
}

let point1 = Point(x: 0, y: 0)
let point2 = Point(x: 3, y: 4)
print("距離: \(point1.distance(to: point2))")

// オプショナル型
var optionalName: String? = "Swift"
if let name = optionalName {
    print("名前: \(name)")
} else {
    print("名前が設定されていません")
}

// 高階関数
let squares = numbers.map { $0 * $0 }
print("二乗: \(squares)")

let evens = numbers.filter { $0 % 2 == 0 }
print("偶数: \(evens)")

let sum = numbers.reduce(0, +)
print("合計: \(sum)")

// プロトコルの定義
protocol Drawable {
    func draw()
}

struct Circle: Drawable {
    var radius: Double
    
    func draw() {
        print("半径\(radius)の円を描画")
    }
}

struct Rectangle: Drawable {
    var width: Double
    var height: Double
    
    func draw() {
        print("\(width)x\(height)の長方形を描画")
    }
}

let shapes: [Drawable] = [
    Circle(radius: 5.0),
    Rectangle(width: 10.0, height: 8.0)
]

for shape in shapes {
    shape.draw()
}

// エラーハンドリング
enum FileError: Error {
    case fileNotFound
    case permissionDenied
}

func readFile(name: String) throws -> String {
    if name.isEmpty {
        throw FileError.fileNotFound
    }
    return "ファイル内容: \(name)"
}

do {
    let content = try readFile(name: "sample.txt")
    print(content)
} catch FileError.fileNotFound {
    print("ファイルが見つかりません")
} catch {
    print("エラー: \(error)")
}

// クロージャ
let greetingClosure = { (name: String) -> String in
    return "Hello, \(name)!"
}

print(greetingClosure("Swift"))
