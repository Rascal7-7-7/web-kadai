// インターフェース
interface Person {
    name: string;
    age: number;
    email?: string; // オプショナルプロパティ
}

interface Greetable {
    greet(): string;
}

// クラスの実装
class Employee implements Person, Greetable {
    constructor(
        public name: string,
        public age: number,
        public department: string,
        public email?: string
    ) {}

    greet(): string {
        return `こんにちは、${this.name}です。${this.department}部門で働いています。`;
    }

    isAdult(): boolean {
        return this.age >= 18;
    }
}

// オブジェクトの作成
const employee: Employee = new Employee("田中太郎", 30, "開発");
console.log(employee.greet());

// ジェネリクス
function identity(arg: T): T {
    return arg;
}

class Stack {
    private items: T[] = [];

    push(item: T): void {
        this.items.push(item);
    }

    pop(): T | undefined {
        return this.items.pop();
    }

    peek(): T | undefined {
        return this.items[this.items.length - 1];
    }

    isEmpty(): boolean {
        return this.items.length === 0;
    }
}

// ジェネリクスの使用
const stringStack = new Stack();
stringStack.push("Hello");
stringStack.push("TypeScript");
console.log(stringStack.pop()); // "TypeScript"

// ユニオン型
type Status = "pending" | "approved" | "rejected";
type ID = string | number;

function processStatus(status: Status): void {
    switch (status) {
        case "pending":
            console.log("処理待ち");
            break;
        case "approved":
            console.log("承認済み");
            break;
        case "rejected":
            console.log("却下");
            break;
    }
}

// 型ガード
function isString(value: any): value is string {
    return typeof value === "string";
}

function processValue(value: string | number): void {
    if (isString(value)) {
        console.log(`文字列: ${value.toUpperCase()}`);
    } else {
        console.log(`数値: ${value.toFixed(2)}`);
    }
}
