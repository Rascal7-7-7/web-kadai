サンプルコード

' Hello World & 基本機能デモ
Module Program
    Sub Main()
        Console.WriteLine("Hello, VB.NET!")

        ' 変数の宣言と出力
        Dim message As String = "VB.NET言語"
        Dim number As Integer = 42
        Dim pi As Double = 3.14159

        Console.WriteLine("{0} - {1} - {2}", message, number, pi)

        ' 配列の使用
        Dim numbers() As Integer = {1, 2, 3, 4, 5}
        For i As Integer = 0 To numbers.Length - 1
            Console.WriteLine("Number[{0}]: {1}", i, numbers(i))
        Next

        ' 関数の呼び出し
        Dim result As Integer = Add(10, 20)
        Console.WriteLine("10 + 20 = {0}", result)

        ' クラスの使用
        Dim person As New Person("田中太郎", 25)
        Console.WriteLine(person.Introduce())

        ' 配列を使った繰り返し
        Dim fruits() As String = {"Apple", "Banana", "Orange"}
        For Each item As String In fruits
            Console.WriteLine(item)
        Next
    End Sub

    ' 関数定義
    Function Add(a As Integer, b As Integer) As Integer
        Return a + b
    End Function

    ' クラス定義（Module内に入れる）
    Class Person
        Private Name As String
        Private Age As Integer

        Public Sub New(name As String, age As Integer)
            Me.Name = name
            Me.Age = age
        End Sub

        Public Function Introduce() As String
            Return String.Format("こんにちは、{0}です。{1}歳です。", Name, Age)
        End Function
    End Class
End Module
